129template <std::size_t N>
130struct n_bytes_property : private boost::totally_ordered<n_bytes_property<N>> {
131 n_bytes_property(property::id id, static_vector<char, N> const& buf)
132 : id_{id},
133 buf_{buf} // very small size copy
134 {
135 }
136
137 template <typename It, typename End>
138 n_bytes_property(property::id id, It b, End e)
139 :id_{id}, buf_(b, e) {}
140
141 n_bytes_property(property::id id, buffer const& buf)
142 :id_{id} {
143 BOOST_ASSERT(buf.size() >= N);
144 buf_.insert(buf_.end(), (buf.begin(), std::next(buf.begin(), N)));
145 }
146
151 std::vector<as::const_buffer> const_buffer_sequence() const {
152 std::vector<as::const_buffer> v;
153 v.reserve(num_of_const_buffer_sequence());
154 v.emplace_back(as::buffer(&id_, 1));
155 v.emplace_back(as::buffer(buf_.data(), buf_.size()));
156 return v;
157 }
158
163 property::id id() const {
164 return id_;
165 }
166
171 std::size_t size() const {
172 return 1 + buf_.size();
173 }
174
179 static constexpr std::size_t num_of_const_buffer_sequence() {
180 return 2;
181 }
182
183 friend bool operator<(n_bytes_property<N> const& lhs, n_bytes_property<N> const& rhs) {
184 return std::tie(lhs.id_, lhs.buf_) < std::tie(rhs.id_, rhs.buf_);
185 }
186
187 friend bool operator==(n_bytes_property<N> const& lhs, n_bytes_property<N> const& rhs) {
188 return std::tie(lhs.id_, lhs.buf_) == std::tie(rhs.id_, rhs.buf_);
189 }
190
191 static constexpr ostream_format const of_ = ostream_format::direct;
192 property::id id_;
193 static_vector<char, N> buf_;
194};

200struct binary_property : private boost::totally_ordered<binary_property> {
201 binary_property(property::id id, buffer buf)
202 :id_{id},
203 buf_{force_move(buf)},
204 length_(2) // size 2
205 {
206 if (buf_.size() > 0xffff) {
207 throw make_error(
208 errc::bad_message,
209 "property::binary_property length is invalid"
210 );
211 }
212 endian_store(boost::numeric_cast<std::uint16_t>(buf_.size()), length_.data());
213 }
214
219 std::vector<as::const_buffer> const_buffer_sequence() const {
220 std::vector<as::const_buffer> v;
221 v.reserve(num_of_const_buffer_sequence());
222 v.emplace_back(as::buffer(&id_, 1));
223 v.emplace_back(as::buffer(length_.data(), length_.size()));
224 v.emplace_back(as::buffer(buf_.data(), buf_.size()));
225 return v;
226 }
227
232 property::id id() const {
233 return id_;
234 }
235
240 std::size_t size() const {
241 return 1 + length_.size() + buf_.size();
242 }
243
248 static constexpr std::size_t num_of_const_buffer_sequence() {
249 return 3;
250 }
251
256 constexpr buffer const& val() const {
257 return buf_;
258 }
259
260 friend bool operator<(binary_property const& lhs, binary_property const& rhs) {
261 return std::tie(lhs.id_, lhs.buf_) < std::tie(rhs.id_, rhs.buf_);
262 }
263
264 friend bool operator==(binary_property const& lhs, binary_property const& rhs) {
265 return std::tie(lhs.id_, lhs.buf_) == std::tie(rhs.id_, rhs.buf_);
266 }
267
268 static constexpr ostream_format const of_ = ostream_format::json_like;
269 property::id id_;
270 buffer buf_;
271 static_vector<char, 2> length_;
272};
